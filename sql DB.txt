CREATE SEQUENCE Reservation_seq 
START WITH 1 
INCREMENT BY 1;

CREATE TABLE Reservations (
    reservation_id NUMBER PRIMARY KEY,
    customer_name VARCHAR2(100),
    reservation_date DATE,
    reservation_time VARCHAR2(100)
);
